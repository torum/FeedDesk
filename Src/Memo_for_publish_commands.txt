
(A) Create package bundle file for Microsoft store.
-------
dotnet publish FeedDesk.csproj /p:Configuration=Release /p:PlatformTarget=x64 /p:Platform=x64 /p:RuntimeIdentifier=win-x64 /p:GenerateAppxPackageOnBuild=true /p:AppxPackageSigningEnabled=false /p:AppxBundle=Never /p:WindowsAppSDKSelfContained=true /p:PublishAot=false -p:WindowsPackageType=MSIX --self-contained true

dotnet publish FeedDesk.csproj /p:Configuration=Release /p:PlatformTarget=x86 /p:Platform=x86 /p:RuntimeIdentifier=win-x86 /p:GenerateAppxPackageOnBuild=true /p:AppxPackageSigningEnabled=false /p:AppxBundle=Never /p:WindowsAppSDKSelfContained=true /p:PublishAot=false -p:WindowsPackageType=MSIX --self-contained true

dotnet publish FeedDesk.csproj /p:Configuration=Release /p:PlatformTarget=ARM64 /p:Platform=ARM64 /p:RuntimeIdentifier=win-arm64 /p:GenerateAppxPackageOnBuild=true /p:AppxPackageSigningEnabled=false /p:AppxBundle=Never /p:WindowsAppSDKSelfContained=true /p:PublishAot=false -p:WindowsPackageType=MSIX --self-contained true

(B) Unpackaged single self-contained exe file.
----
dotnet publish -c Release -r win-x64 -p:PublishAot=false -p:PublishSingleFile=true -p:WindowsPackageType=None -p:AppxPackage=false -p:EnableMsixTooling=true --self-contained true
----

(C) Unpackaged Aot compiled.
----
dotnet publish -c Release -r win-x64 -p:PublishAot=true -p:WindowsPackageType=None -p:AppxPackage=false -p:EnableMsixTooling=false --self-contained true
Or simply
dotnet publish -c Release -r win-x64 -p:PublishAot=true --self-contained true
